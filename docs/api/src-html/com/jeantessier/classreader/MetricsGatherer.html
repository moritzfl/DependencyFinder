253    // Classfile
254    public void visitClassfile(Classfile classfile) {
255        if (classfile.isPublic()) {
256            publicClasses.add(classfile);
257        } else {
258            packageClasses.add(classfile);
259        }
260
261        if (classfile.isFinal()) {
262            finalClasses.add(classfile);
263        }
264
265        if (classfile.isInterface()) {
266            interfaces.add(classfile);
267        } else {
268            classes.add(classfile);
269        }
270
271        if (classfile.isAbstract()) {
272            abstractClasses.add(classfile);
273        }
274
275        super.visitClassfile(classfile);
276    }

278    // Features
279    public void visitField_info(Field_info entry) {
280        fields.add(entry);
281
282        if (entry.isPublic()) {
283            publicFields.add(entry);
284        } else if (entry.isPrivate()) {
285            privateFields.add(entry);
286        } else if (entry.isProtected()) {
287            protectedFields.add(entry);
288        } else {
289            packageFields.add(entry);
290        }
291
292        if (entry.isStatic()) {
293            staticFields.add(entry);
294        }
295
296        if (entry.isFinal()) {
297            finalFields.add(entry);
298        }
299
300        if (entry.isVolatile()) {
301            volatileFields.add(entry);
302        }
303
304        if (entry.isTransient()) {
305            transientFields.add(entry);
306        }
307
308        super.visitField_info(entry);
309    }
310

373    public void visitSynthetic_attribute(Synthetic_attribute attribute) {
374        super.visitSynthetic_attribute(attribute);
375        visitAttribute("Synthetic");
376
377        Object owner = attribute.getOwner();
378
379        if (owner instanceof Classfile) {
380            syntheticClasses.add((Classfile) owner);
381        } else if (owner instanceof Field_info) {
382            syntheticFields.add((Field_info) owner);
383        } else if (owner instanceof Method_info) {
384            syntheticMethods.add((Method_info) owner);
385        } else {
386            Logger.getLogger(getClass()).warn("Synthetic attribute on unknown Visitable: " + owner.getClass().getName());
387        }
388    }

420    public void visitDeprecated_attribute(Deprecated_attribute attribute) {
421        super.visitDeprecated_attribute(attribute);
422        visitAttribute("Deprecated");
423
424        Object owner = attribute.getOwner();
425
426        if (owner instanceof Classfile) {
427            deprecatedClasses.add((Classfile) owner);
428        } else if (owner instanceof Field_info) {
429            deprecatedFields.add((Field_info) owner);
430        } else if (owner instanceof Method_info) {
431            deprecatedMethods.add((Method_info) owner);
432        } else {
433            Logger.getLogger(getClass()).warn("Deprecated attribute on unknown Visitable: " + owner.getClass().getName());
434        }
435    }

437    public void visitCustom_attribute(Custom_attribute attribute) {
438        super.visitCustom_attribute(attribute);
439        visitAttribute("custom");
440        customAttributes.add(attribute);
441    }
442
443    private void visitAttribute(String attributeName) {
444        attributeCounts.put(attributeName, attributeCounts.get(attributeName) + 1);
445    }
446
447    // Attribute helpers
448    public void visitInstruction(Instruction helper) {
449        getInstructionCounts()[helper.getOpcode()]++;
450
451        super.visitInstruction(helper);
452    }

455        if (helper.isPublic()) {

458            privateInnerClasses.add(helper);
459        } else if (helper.isProtected()) {

462            packageInnerClasses.add(helper);

465        if (helper.isStatic()) {

477        }

482    }
