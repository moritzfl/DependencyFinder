033package com.jeantessier.classreader;
034
035import java.util.*;
036
037public interface Classfile extends Deprecatable, Visitable {
038    public ClassfileLoader getLoader();
039
040    public int getMagicNumber();
041    public int getMinorVersion();
042    public int getMajorVersion();
043
044    public ConstantPool getConstantPool();
045
046    public int getAccessFlag();
047
048    public int getClassIndex();
049    public Class_info getRawClass();
050    public String getClassName();
051    public String getPackageName();
052    public String getSimpleName();
053
054    public int getSuperclassIndex();
055    public Class_info getRawSuperclass();
056    public String getSuperclassName();
057
058    public Collection<? extends Class_info> getAllInterfaces();
059    public Class_info getInterface(String name);
060
061    public Collection<? extends Field_info> getAllFields();
062    public Field_info getField(String name);
063    public Field_info locateField(String name);
064
065    public Collection<? extends Method_info> getAllMethods();
066    public Method_info getMethod(String signature);
067    public Method_info locateMethod(String signature);
068
069    public Collection<? extends Attribute_info> getAttributes();
070
071    public boolean isPublic();
072    public boolean isPackage();
073    public boolean isFinal();
074    public boolean isSuper();
075    public boolean isInterface();
076    public boolean isAbstract();
077    public boolean isAnnotation();
078    public boolean isEnum();
079    public boolean isSynthetic();
080    public boolean isDeprecated();
081    public boolean isGeneric();
082
083    public String getDeclaration();
084
085    public boolean isInnerClass();
086    public boolean isMemberClass();
087    public boolean isLocalClass();
088    public boolean isAnonymousClass();
089}
