610    public int getLength() {
611        int result = length[getOpcode()];
612
613        int padding, low, high, npairs;
614    
615        switch (getOpcode()) {
616            case 0xaa:
617                // tableswitch
618                padding = 3 - (start % 4);
619                low =
620                    (getByte(padding+5) << 24) |
621                    (getByte(padding+6) << 16) |
622                    (getByte(padding+7) << 8) |
623                    (getByte(padding+8));
624                high =
625                    (getByte(padding+9) << 24) |
626                    (getByte(padding+10) << 16) |
627                    (getByte(padding+11) << 8) |
628                    (getByte(padding+12));
629                result =
630                    1 +                   // opcode
631                    padding +             // padding
632                    12 +                  // default + low + high
633                    (high - low + 1) * 4; // (high - low + 1) * offset
634                break;
635
636            case 0xab:
637                // lookupswitch
638                padding = 3 - (start % 4);
639                npairs =
640                    (getByte(padding+5) << 24) |
641                    (getByte(padding+6) << 16) |
642                    (getByte(padding+7) << 8) |
643                    (getByte(padding+8));
644                result =
645                    1 +            // opcode
646                    padding +      // padding
647                    8 +            // default + npairs
648                    (npairs * 8);  // npairs * (match + offset)
649                break;
650
651            case 0xc4:
652                // wide
653                if (getByte(1) == 0x84 /* iinc */) {
654                    result = 6;
655                } else {
656                    result = 4;
657                }
658                break;
659
660            default:
661                // Do nothing
662                break;
663        }
664
665        return result;
666    }

995        boolean result;
996
997        if (this == object) {
998            result = true;
999        } else if (object == null || getClass() != object.getClass()) {
1000            result = false;
1001        } else {
1002            Instruction other = (Instruction) object;
1003            result = getOpcode() == other.getOpcode();
1004
1005            ConstantPoolEntry thisEntry = (ConstantPoolEntry) ((code != null) ? getIndexedConstantPoolEntry() : null);
1006            ConstantPoolEntry otherEntry = (ConstantPoolEntry) ((other.code != null) ? other.getIndexedConstantPoolEntry() : null);
1007
1008            if (result && thisEntry != null && otherEntry != null) {
1009                result = thisEntry.equals(otherEntry);
1010            } else {
1011                for (int i=1; result && i<getLength(); i++) {
1012                    result = bytecode[start+i] == other.bytecode[other.start+i];
1013                }
1014            }
1015        }
1016
1017        return result;
