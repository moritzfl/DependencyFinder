077    public Classfile(ClassfileLoader loader, DataInput in, AttributeFactory attributeFactory) throws IOException {
078        this.loader = loader;
079
080        magicNumber = in.readInt();
081        Logger.getLogger(getClass()).debug("magic number = 0x" + Integer.toHexString(magicNumber).toUpperCase());
082
083        if (magicNumber != 0xCAFEBABE) {
084            throw new IOException("Bad magic number");
085        }
086        
087        // Reading the file format's version number
088        minorVersion = in.readUnsignedShort();
089        Logger.getLogger(getClass()).debug("minor version = " + minorVersion);
090        majorVersion = in.readUnsignedShort();
091        Logger.getLogger(getClass()).debug("major version = " + majorVersion);
092
093        // Reading the constant pool
094        Logger.getLogger(getClass()).debug("Reading the constant pool ...");
095        constantPool = new ConstantPool(this, in);
096        Logger.getLogger(getClass()).debug(constantPool);
097
098        // Skipping the access flag
099        accessFlag = in.readUnsignedShort();
100        Logger.getLogger(getClass()).debug("accessFlag = " + accessFlag);
101
102        // Retrieving this class's name
103        classIndex = in.readUnsignedShort();
104        Logger.getLogger(getClass()).debug("thisClass = " + classIndex + " (" + getClassName() + ")");
105
106        // Retrieving this class's superclass
107        superclassIndex = in.readUnsignedShort();
108        Logger.getLogger(getClass()).debug("superclass = " + superclassIndex + " (" + getSuperclassName() + ")");
109
110        // Retrieving the inferfaces
111        int interfaceCount = in.readUnsignedShort();
112        Logger.getLogger(getClass()).debug("Reading " + interfaceCount + " interface(s)");
113        for (int i=0; i<interfaceCount; i++) {
114            Class_info interfaceInfo = (Class_info) constantPool.get(in.readUnsignedShort());
115            Logger.getLogger(getClass()).debug("    " + interfaceInfo.getName());
116            interfaces.add(interfaceInfo);
117        }
118
119        // Retrieving the fields
120        int fieldCount = in.readUnsignedShort();
121        Logger.getLogger(getClass()).debug("Reading " + fieldCount + " field(s)");
122        for (int i=0; i<fieldCount; i++) {
123            Logger.getLogger(getClass()).debug("Field " + i + ":");
124            fields.add(new Field_info(this, in));
125        }
126
127        // Retrieving the methods
128        int methodCount = in.readUnsignedShort();
129        Logger.getLogger(getClass()).debug("Reading " + methodCount + " method(s)");
130        for (int i=0; i<methodCount; i++) {
131            Logger.getLogger(getClass()).debug("Method " + i + ":");
132            methods.add(new Method_info(this, in));
133        }
134
135        // Retrieving the attributes
136        int attributeCount = in.readUnsignedShort();
137        Logger.getLogger(getClass()).debug("Reading " + attributeCount + " class attribute(s)");
138        for (int i=0; i<attributeCount; i++) {
139            Logger.getLogger(getClass()).debug("Attribute " + i + ":");
140            attributes.add(attributeFactory.create(constantPool, this, in));
141        }
142    }

147    Classfile(ClassfileLoader loader, ConstantPool constantPool, int accessFlag, int classIndex, int superclassIndex, Iterable<Class_info> interfaces, Iterable<Field_info> fields, Iterable<Method_info> methods, Iterable<Attribute_info> attributes) {
148        this.loader = loader;

158        for (Field_info field : fields) {

160        }
161
162        for (Method_info method : methods) {
163            this.methods.add(method);
164        }
165
166        for (Attribute_info attribute : attributes) {
167            this.attributes.add(attribute);
168        }
169    }

261    public com.jeantessier.classreader.Field_info locateField(String name) {
262        com.jeantessier.classreader.Field_info localField = getField(name);
263        if (localField != null) {
264            return localField;
265        }
266
267        com.jeantessier.classreader.Classfile superclass = getLoader().getClassfile(getSuperclassName());
268        if (superclass != null) {
269            com.jeantessier.classreader.Field_info inheritedField = superclass.locateField(name);
270            if (inheritedField != null && (inheritedField.isPublic() || inheritedField.isProtected() || (inheritedField.isPackage() && inheritedField.getClassfile().getPackageName().equals(superclass.getPackageName())))) {
271                return inheritedField;
272            }
273        }
274
275
276        for (com.jeantessier.classreader.Class_info interfaceInfo : getAllInterfaces()) {
277            com.jeantessier.classreader.Classfile interfaceClassfile = getLoader().getClassfile(interfaceInfo.getName());
278            if (interfaceClassfile != null) {
279                com.jeantessier.classreader.Field_info interfaceField = interfaceClassfile.locateField(name);
280                if (interfaceField != null && (interfaceField.isPublic() || interfaceField.isProtected())) {
281                    return interfaceField;
282                }
283            }
284        }
285
286        return null;
287    }

303    public com.jeantessier.classreader.Method_info locateMethod(String signature) {
304        com.jeantessier.classreader.Method_info localMethod = getMethod(signature);
305        if (localMethod != null) {
306            return localMethod;
307        }
308
309        com.jeantessier.classreader.Classfile superclass = getLoader().getClassfile(getSuperclassName());
310        if (superclass != null) {
311            com.jeantessier.classreader.Method_info inheritedMethod = superclass.locateMethod(signature);
312            if (inheritedMethod != null && (inheritedMethod.isPublic() || inheritedMethod.isProtected() || (inheritedMethod.isPackage() && inheritedMethod.getClassfile().getPackageName().equals(superclass.getPackageName())))) {
313                return inheritedMethod;
314            }
315        }
316
317        for (com.jeantessier.classreader.Class_info inferfaceInfo : getAllInterfaces()) {
318            com.jeantessier.classreader.Classfile interfaceClassfile = getLoader().getClassfile(inferfaceInfo.getName());
319            if (interfaceClassfile != null) {
320                com.jeantessier.classreader.Method_info interfaceMethod = interfaceClassfile.locateMethod(signature);
321                if (interfaceMethod != null && (interfaceMethod.isPublic() || interfaceMethod.isProtected())) {
322                    return interfaceMethod;
323                }
324            }
325        }
326
327        return null;
328    }

374    private boolean isSyntheticFromAttribute() {
375        boolean result = false;
376
377        Iterator i = getAttributes().iterator();
378        while (!result && i.hasNext()) {
379            result = i.next() instanceof Synthetic_attribute;
380        }
381
382        return result;
383    }

407    public String getDeclaration() {
408        StringBuffer result = new StringBuffer();
409
410        if (isPublic()) result.append("public ");
411        if (isFinal()) result.append("final ");
412
413        if (isInterface()) {
414            result.append("interface ").append(getClassName());
415
416            if (getAllInterfaces().size() != 0) {
417                result.append(" extends ");
418                Iterator i = getAllInterfaces().iterator();
419                while (i.hasNext()) {
420                    result.append(i.next());
421                    if (i.hasNext()) {
422                        result.append(", ");
423                    }
424                }
425            }
426        } else {
427            if (isAbstract()) result.append("abstract ");
428            result.append("class ").append(getClassName());
429
430            if (getSuperclassIndex() != 0) {
431                result.append(" extends ").append(getSuperclassName());
432            }
433            
434            if (getAllInterfaces().size() != 0) {
435                result.append(" implements ");
436                Iterator i = getAllInterfaces().iterator();
437                while (i.hasNext()) {
438                    result.append(i.next());
439                    if (i.hasNext()) {
440                        result.append(", ");
441                    }
442                }
443            }
444        }
445
446        return result.toString();
447    }

461    public boolean isMemberClass() {
462        boolean result = false;
463
464        InnerClass innerClass = getMatchingInnerClass();
465        if (innerClass != null) {
466            result = innerClass.isMemberClass();
467        }
468
469        return result;

472    public boolean isLocalClass() {
473        boolean result = false;
474
475        InnerClass innerClass = getMatchingInnerClass();
476        EnclosingMethod_attribute enclosingMethod = getEnclosingMethod();
477        if (innerClass != null && enclosingMethod != null) {
478            result = !innerClass.isAnonymousClass();
479        }
480
481        return result;
482    }
483
484    public boolean isAnonymousClass() {
485        boolean result = false;
486
487        InnerClass innerClass = getMatchingInnerClass();
488        if (innerClass != null) {
489            result = innerClass.isAnonymousClass();
490        }
491
492        return result;
493    }
494
495    private InnerClass getMatchingInnerClass() {
496        InnerClass result = null;
497
498        for (Attribute_info attribute : getAttributes()) {
499            if (attribute instanceof InnerClasses_attribute) {
500                for (InnerClass innerClass : ((InnerClasses_attribute) attribute).getInnerClasses()) {
501                    if (innerClass.getInnerClassInfo().equals(getClassName())) {
502                        result = innerClass;
503                    }
504                }
505            }
506        }
507
508        return result;
509    }
510
511    private EnclosingMethod_attribute getEnclosingMethod() {
512        EnclosingMethod_attribute result = null;
513
514        for (Attribute_info attribute : getAttributes()) {
515            if (attribute instanceof EnclosingMethod_attribute) {
516                result = (EnclosingMethod_attribute) attribute;
517            }
518        }
519
520        return result;
521    }
