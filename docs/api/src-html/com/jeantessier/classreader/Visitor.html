033package com.jeantessier.classreader;
034
035import java.util.Collection;
036
037public interface Visitor {
038    // Classfile
039    public void visitClassfiles(Collection<Classfile> classfiles);
040    public void visitClassfile(Classfile classfile);
041
042    // ConstantPool entries
043    public void visitConstantPool(ConstantPool constantPool);
044    public void visitClass_info(Class_info entry);
045    public void visitFieldRef_info(FieldRef_info entry);
046    public void visitMethodRef_info(MethodRef_info entry);
047    public void visitInterfaceMethodRef_info(InterfaceMethodRef_info entry);
048    public void visitString_info(String_info entry);
049    public void visitInteger_info(Integer_info entry);
050    public void visitFloat_info(Float_info entry);
051    public void visitLong_info(Long_info entry);
052    public void visitDouble_info(Double_info entry);
053    public void visitNameAndType_info(NameAndType_info entry);
054    public void visitUTF8_info(UTF8_info entry);
055    public void visitMethodHandle_info(MethodHandle_info entry);
056    public void visitMethodType_info(MethodType_info entry);
057    public void visitInvokeDynamic_info(InvokeDynamic_info entry);
058
059    // Features
060    public void visitField_info(Field_info entry);
061    public void visitMethod_info(Method_info entry);
062
063    // Attributes
064    public void visitConstantValue_attribute(ConstantValue_attribute attribute);
065    public void visitCode_attribute(Code_attribute attribute);
066    public void visitExceptions_attribute(Exceptions_attribute attribute);
067    public void visitInnerClasses_attribute(InnerClasses_attribute attribute);
068    public void visitEnclosingMethod_attribute(EnclosingMethod_attribute attribute);
069    public void visitSynthetic_attribute(Synthetic_attribute attribute);
070    public void visitSignature_attribute(Signature_attribute attribute);
071    public void visitSourceFile_attribute(SourceFile_attribute attribute);
072    public void visitSourceDebugExtension_attribute(SourceDebugExtension_attribute attribute);
073    public void visitLineNumberTable_attribute(LineNumberTable_attribute attribute);
074    public void visitLocalVariableTable_attribute(LocalVariableTable_attribute attribute);
075    public void visitLocalVariableTypeTable_attribute(LocalVariableTypeTable_attribute attribute);
076    public void visitDeprecated_attribute(Deprecated_attribute attribute);
077    public void visitRuntimeVisibleAnnotations_attribute(RuntimeVisibleAnnotations_attribute attribute);
078    public void visitRuntimeInvisibleAnnotations_attribute(RuntimeInvisibleAnnotations_attribute attribute);
079    public void visitRuntimeVisibleParameterAnnotations_attribute(RuntimeVisibleParameterAnnotations_attribute attribute);
080    public void visitRuntimeInvisibleParameterAnnotations_attribute(RuntimeInvisibleParameterAnnotations_attribute attribute);
081    public void visitAnnotationDefault_attribute(AnnotationDefault_attribute attribute);
082    public void visitCustom_attribute(Custom_attribute attribute);
083
084    // Attribute helpers
085    public void visitInstruction(Instruction instruction);
086    public void visitExceptionHandler(ExceptionHandler helper);
087    public void visitInnerClass(InnerClass helper);
088    public void visitLineNumber(LineNumber helper);
089    public void visitLocalVariable(LocalVariable helper);
090    public void visitLocalVariableType(LocalVariableType helper);
091
092    // Annotations
093    public void visitParameter(Parameter helper);
094    public void visitAnnotation(Annotation helper);
095    public void visitElementValuePair(ElementValuePair helper);
096    public void visitByteConstantElementValue(ByteConstantElementValue helper);
097    public void visitCharConstantElementValue(CharConstantElementValue helper);
098    public void visitDoubleConstantElementValue(DoubleConstantElementValue helper);
099    public void visitFloatConstantElementValue(FloatConstantElementValue helper);
100    public void visitIntegerConstantElementValue(IntegerConstantElementValue helper);
101    public void visitLongConstantElementValue(LongConstantElementValue helper);
102    public void visitShortConstantElementValue(ShortConstantElementValue helper);
103    public void visitBooleanConstantElementValue(BooleanConstantElementValue helper);
104    public void visitStringConstantElementValue(StringConstantElementValue helper);
105    public void visitEnumElementValue(EnumElementValue helper);
106    public void visitClassElementValue(ClassElementValue helper);
107    public void visitAnnotationElementValue(AnnotationElementValue helper);
108    public void visitArrayElementValue(ArrayElementValue helper);
109}
