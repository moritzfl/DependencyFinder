033package com.jeantessier.diff;
034
035import java.util.*;
036
037import org.apache.log4j.*;
038
039/**
040 *  Documents the difference, if any, for a given package.
041 */
042public class PackageDifferences extends RemovableDifferences {
043    private Collection<Differences> classDifferences = new LinkedList<Differences>();
044
045    private String oldDeclaration = null;
046    private String newDeclaration = null;
047
048    /**
049     *  Only the DifferencesFactory can create instances of this class.
050     */
051    PackageDifferences(String name, Map oldPackage, Map newPackage) {
052        super(name);
053
054        if (oldPackage != null && !oldPackage.isEmpty()) {
055            oldDeclaration = name;
056        }
057
058        if (newPackage != null && !newPackage.isEmpty()) {
059            newDeclaration = name;
060        }
061    
062        if (isModified()) {
063            Logger.getLogger(getClass()).debug(getName() + " declaration has been modified.");
064        } else {
065            Logger.getLogger(getClass()).debug(getName() + " declaration has not been modified.");
066        }
067    }
068
069    public Collection<Differences> getClassDifferences() {
070        return classDifferences;
071    }
072
073    public String getOldDeclaration() {
074        return oldDeclaration;
075    }
076
077    public String getNewDeclaration() {
078        return newDeclaration;
079    }
080
081    public boolean isModified() {
082        return super.isModified() || (getClassDifferences().size() != 0);
083    }
084
085    public void accept(Visitor visitor) {
086        visitor.visitPackageDifferences(this);
087    }
088}
